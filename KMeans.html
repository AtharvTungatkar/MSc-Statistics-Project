    silhouette_scores.append(silhouette_score(df, cluster_labels[-1]))

# Find the optimal number of clusters with the highest silhouette score
optimal_clusters = cluster_range[np.argmax(silhouette_scores)]

    p=(clustering_results.T[asset_of_interest]==clustering_results.T[companies[i]]).value_counts()[1]/len(clustering_results.T)
